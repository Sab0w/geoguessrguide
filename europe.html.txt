<script>
(function() {
  var countries = [];

  function saveToCookie() {
    try {
      var data = JSON.stringify(countries);
      var expires = new Date();
      expires.setFullYear(expires.getFullYear() + 1);
      document.cookie = "geoguessr_data=" + encodeURIComponent(data) + "; expires=" + expires.toUTCString() + "; path=/";
      return true;
    } catch(e) {
      console.error("Erreur sauvegarde:", e);
      return false;
    }
  }

  function loadFromCookie() {
    try {
      var cookies = document.cookie.split(';');
      for (var i = 0; i < cookies.length; i++) {
        var cookie = cookies[i].trim();
        if (cookie.indexOf("geoguessr_data=") === 0) {
          var data = decodeURIComponent(cookie.substring(15));
          countries = JSON.parse(data);
          return true;
        }
      }
    } catch(e) {
      console.error("Erreur chargement:", e);
    }
    return false;
  }

  function render() {
    var grid = document.getElementById("grid");
    var select = document.getElementById("countrySelect");
    var deleteBtn = document.getElementById("deleteBtn");
    
    grid.innerHTML = "";
    select.innerHTML = '<option value="">-- S√©lectionner un pays √† supprimer --</option>';
    
    if (countries.length === 0) {
      grid.innerHTML = '<div class="empty-message">Aucun pays ajout√©. Cliquez sur "‚ûï Ajouter" !</div>';
      deleteBtn.disabled = true;
      return;
    }
    
    deleteBtn.disabled = false;
    
    for (var i = 0; i < countries.length; i++) {
      var c = countries[i];
      
      var card = document.createElement("a");
      card.className = "country-card";
      card.href = c.link || "#";
      if (c.link && c.link !== "#") card.target = "_blank";
      
      var flagDiv = document.createElement("div");
      flagDiv.className = "flag-header";
      flagDiv.style.backgroundImage = "url('" + (c.flag || "https://via.placeholder.com/300x150") + "')";
      
      var nameDiv = document.createElement("div");
      nameDiv.className = "country-name";
      nameDiv.textContent = c.name;
      flagDiv.appendChild(nameDiv);
      
      var contentDiv = document.createElement("div");
      contentDiv.className = "card-content";
      
      var metaRow = document.createElement("div");
      metaRow.className = "hint-row";
      metaRow.innerHTML = '<span class="tag meta">M√©ta</span><span>' + (c.meta || "-") + '</span>';
      
      var landRow = document.createElement("div");
      landRow.className = "hint-row";
      landRow.innerHTML = '<span class="tag landscape">Paysage</span><span>' + (c.land || "-") + '</span>';
      
      contentDiv.appendChild(metaRow);
      contentDiv.appendChild(landRow);
      
      card.appendChild(flagDiv);
      card.appendChild(contentDiv);
      grid.appendChild(card);
      
      var opt = document.createElement("option");
      opt.value = i;
      opt.textContent = c.name;
      select.appendChild(opt);
    }
  }

  document.getElementById("addBtn").onclick = function() {
    document.getElementById("myModal").className = "modal show";
  };

  document.getElementById("closeBtn").onclick = function() {
    document.getElementById("myModal").className = "modal";
  };

  document.getElementById("myModal").onclick = function(e) {
    if (e.target === this) {
      this.className = "modal";
    }
  };

  document.getElementById("saveBtn").onclick = function() {
    var name = document.getElementById("inputName").value.trim();
    
    if (!name) {
      alert("Le nom est obligatoire !");
      return;
    }
    
    var country = {
      id: Date.now(),
      name: name,
      flag: document.getElementById("inputFlag").value.trim(),
      meta: document.getElementById("inputMeta").value.trim(),
      land: document.getElementById("inputLandscape").value.trim(),
      link: document.getElementById("inputLink").value.trim()
    };
    
    countries.push(country);
    saveToCookie();
    render();
    
    document.getElementById("myModal").className = "modal";
    document.getElementById("inputName").value = "";
    document.getElementById("inputFlag").value = "";
    document.getElementById("inputMeta").value = "";
    document.getElementById("inputLandscape").value = "";
    document.getElementById("inputLink").value = "";
  };

  document.getElementById("deleteBtn").onclick = function() {
    var select = document.getElementById("countrySelect");
    var idx = parseInt(select.value);
    
    if (isNaN(idx) || idx < 0) {
      alert("S√©lectionnez un pays √† supprimer");
      return;
    }
    
    if (idx >= countries.length) {
      alert("Erreur: pays introuvable");
      return;
    }
    
    var name = countries[idx].name;
    
    if (confirm("Supprimer \"" + name + "\" ?")) {
      countries.splice(idx, 1);
      saveToCookie();
      render();
    }
  };

  loadFromCookie();
  render();
})();
